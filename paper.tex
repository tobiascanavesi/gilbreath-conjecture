\begin{abstract}
In this paper, we propose a conditional framework for Gilbreath's conjecture, which states that the first element of each row of the absolute difference 
triangle of primes is always~$1$, by presenting a new mechanism called \emph{XOR bias absorption}. 
Using Odlyzko~\cite{Odlyzko1993} reduction mod~4 and the random analogue of Chase~\cite{Chase2020}, we show that the parity
sequence $b(n) = \lfloor d_1(n)/2 \rfloor \bmod 2$, which is constructed from prime gaps, carries a systematic bias of
magnitude $\varepsilon(x) \approx C/(\log x)^{\alpha}$ with $\alpha$ estimated between $0.87$ (raw regression)
and $1.0$ (product diagnostic), derived from the Hardy--Littlewood singular series. 
We prove exactly (Theorem~\ref{thm:xor-absorption}) that sums of XOR of independent biased coins converge exponentially fast to fairness, 
and verify computationally that the prime gap parities satisfy the requisite weak-mixing conditions (asymptotic $2$-dependence).
Assuming three explicit hypotheses---the Hardy--Littlewood conjecture, quantitative $2$-dependence, and binary absorption for primes---we
prove that Gilbreath's conjecture is true for all sufficiently large~$k$.
We also prove two unconditional results toward binary absorption: a deterministic one-step shrinkage bound,
and full absorption for any sequence with gaps in $\{2, 4\}$.
All intermediate results have been computationally verified for primes up to $10^9$, with samples of up to $10^5$ primes per scale.
\end{abstract}

\section{Introduction}
\label{sec:intro}
% ====================================================================

Let $p_1 = 2, p_2 = 3, p_3 = 5, \ldots$ denote the primes in their natural ordering. Following Odlyzko~\cite{Odlyzko1993}, define
\begin{equation}\label{eq:gilbreath-def}
d_0(n) = p_n, \qquad d_{k+1}(n) = |d_k(n) - d_k(n+1)| \quad (k \ge 0,\; n \ge 1).
\end{equation}
The sequence $(d_k(1))_{k \ge 0} = 2, 1, 1, 1, \ldots$ constitutes the first column of the \emph{Gilbreath triangle}.

\begin{conjecture}[Gilbreath, 1958; Proth~\cite{Proth1878}, 1878]
$d_k(1) = 1$ for all $k \ge 1$.
\end{conjecture}

This conjecture has been verified computationally by Odlyzko~\cite{Odlyzko1993} for all $k \le \pi(10^{13}) \approx 3.46 \times 10^{11}$. 
Chase~\cite{Chase2020} proved a random analogue: for sequences beginning with~$2$ followed by increasing odd numbers with random gaps 
bounded by $O(\log\log M / \log\log\log M)$, the Gilbreath property holds almost surely. 
However, actual primes have gaps of order $O(\log p)$ on average, with irregular fluctuations in short intervals~\cite{Maier1985}, far exceeding Chase's bound.

\subsection{Our contribution}
We identify a three-step mechanism explaining why Gilbreath's conjecture holds:
\begin{enumerate}[(1)]
\item \textbf{Mod~4 reduction (Odlyzko):} The Gilbreath triangle, reduced modulo~4, follows XOR dynamics on binary sequences derived from prime gaps.
\item \textbf{Hardy-Littlewood bias:} These binary sequences carry a computable bias $\varepsilon(x)$ from the singular series, which decays as $O(1/(\log x)^{\alpha})$.
\item \textbf{XOR bias absorption (new application):} XOR sums of many weakly-dependent biased bits converge exponentially fast to fair coin flips, explaining why the first column remains~$1$.
\end{enumerate}

We formalize this as a conditional theorem (Theorem~\ref{thm:main}), identifying three explicit hypotheses that together imply Gilbreath's 
conjecture for all sufficiently large~$k$.

\section{Definitions and the mod~4 framework}
\label{sec:mod4}
% ====================================================================

\begin{definition}[Binary parity sequence]\label{def:b-sequence}
For $n \ge 2$, define
\[
b(n) = \left\lfloor \frac{d_1(n)}{2} \right\rfloor \bmod 2,
\]
where $d_1(n) = p_{n+1} - p_n$ is the $n$-th prime gap. Since $p_n$ and $p_{n+1}$ are both odd for $n \ge 2$, $d_1(n)$ is even, 
and $b(n) \in \{0, 1\}$ is well-defined. Equivalently, $b(n) = 0$ if $d_1(n) \equiv 0 \pmod{4}$, and $b(n) = 1$ if $d_1(n) \equiv 2 \pmod{4}$.
\end{definition}

\begin{proposition}[Odlyzko {\cite[Section~2]{Odlyzko1993}}]\label{prop:mod4}
For $k \ge 1$ and $n \ge 2$:
\begin{enumerate}[(a)]
\item $d_k(n)$ is even.
\item $d_{k+1}(n) \equiv d_k(n) + d_k(n+1) \pmod{4}$.
\end{enumerate}
\end{proposition}

\begin{proof}
Part~(a) follows by induction on $k$: the base case $d_1(n) = p_{n+1} - p_n$ is even for $n \ge 2$ (both primes odd), 
and $|{\rm even} - {\rm even}| = {\rm even}$.

For part~(b), write $d_k(n) = 2a$ and $d_k(n+1) = 2c$ with $a, c \ge 0$. Then
\[
d_{k+1}(n) = |2a - 2c| = 2|a - c|.
\]
Since $|a - c| = a + c - 2\min(a,c)$, we have $|a-c| \equiv a + c \pmod{2}$, hence
\[
2|a-c| \equiv 2(a+c) = d_k(n) + d_k(n+1) \pmod{4}. \qedhere
\]
\end{proof}

\begin{corollary}[XOR reduction]\label{cor:xor}
Define $e_k(n) = (d_k(n)/2) \bmod 2$ for $k \ge 1$, $n \ge 2$. Then
\[
e_{k+1}(n) = e_k(n) \oplus e_k(n+1),
\]
where $\oplus$ denotes addition modulo~$2$. In particular, $e_1(n) = b(n)$.
\end{corollary}

\begin{proof}
Dividing the congruence in Proposition~\ref{prop:mod4}(b) by~$2$:
\[
\frac{d_{k+1}(n)}{2} \equiv \frac{d_k(n)}{2} + \frac{d_k(n+1)}{2} \pmod{2}. \qedhere
\]
\end{proof}

\begin{remark}
Corollary~\ref{cor:xor} reduces the mod~4 dynamics of the Gilbreath triangle (for $n \ge 2$) to the XOR triangle, equivalently, Pascal's triangle 
modulo~2 is applied to the binary sequence $(b(n))_{n \ge 2}$. This was computationally verified for $2 \times 10^6$ instances ($10^5$ primes $\times$ $20$ rows) with zero violations.
\end{remark}

\section{The first column mechanism}
\label{sec:first-column}
% ====================================================================

\begin{proposition}[Parity of the first column]\label{prop:odd}
For all $k \ge 1$, $d_k(1)$ is odd.
\end{proposition}

\begin{proof}
$d_1(1) = |p_2 - p_1| = 1$, which is odd. Inductively, $d_k(1) = |d_{k-1}(1) - d_{k-1}(2)|$. By induction $d_{k-1}(1)$ is odd, 
and by Proposition~\ref{prop:mod4}(a), $d_{k-1}(2)$ is even (for $k \ge 2$). Hence $|{\rm odd} - {\rm even}|$ is odd.
\end{proof}

\begin{proposition}[Reduction to $d_k(2)$]\label{prop:reduction}
Suppose $d_k(2) \in \{0, 2\}$ for all $k \ge K$. Then:
\begin{enumerate}[(a)]
\item The sequence $d_k(1)$ is non-increasing for $k \ge K$. In particular, $d_k(1) \in \{1, 3\}$ for all $k \ge K'$
for some finite $K' \ge K$; specifically, at most $\lfloor(d_K(1) - 1)/2\rfloor$ occurrences of $d_k(2) = 2$ are needed.
\item If $d_k(1) \in \{1,3\}$ and $d_k(2) = 2$, then $d_{k+1}(1) = 1$ (reset pulse).
\item If $d_k(2) = 0$, then $d_{k+1}(1) = d_k(1)$ (the value is preserved).
\end{enumerate}
\end{proposition}

\begin{proof}
Since $d_k(1)$ is odd (Proposition~\ref{prop:odd}) and $d_k(2) \in \{0, 2\}$:
\begin{itemize}
\item $d_k(2) = 0$: $d_{k+1}(1) = |d_k(1) - 0| = d_k(1)$, proving~(c).
\item $d_k(2) = 2$: $d_{k+1}(1) = |d_k(1) - 2|$. Since $d_k(1) \ge 1$ is odd, this equals $d_k(1) - 2$ when $d_k(1) \ge 3$, and $1$ when $d_k(1) = 1$.
\end{itemize}
For part~(a), observe that in both cases $d_{k+1}(1) \le d_k(1)$. Moreover, each occurrence of $d_k(2) = 2$ with $d_k(1) \ge 3$ \emph{strictly} 
decreases $d_k(1)$ by~$2$. Starting from any odd value $d_K(1) = 2m+1$, at most $m$ such resets bring $d_k(1)$ down to~$1$, which is then
permanently stable (since $|1-2| = 1$ and $|1-0| = 1$). In particular, $d_k(1) \in \{1, 3\}$ after at most $m = \lfloor(d_K(1)-1)/2\rfloor$
occurrences of $d_k(2) = 2$ past row~$K$, proving~(a). Once $d_k(1) \in \{1,3\}$, a reset gives $|1-2|=1$ or $|3-2|=1$, proving~(b).
\end{proof}

\begin{remark}\label{rem:heartbeat}
Proposition~\ref{prop:reduction} reveals a ``heartbeat'' mechanism: once binary absorption is achieved ($d_k(2) \in \{0,2\}$), 
the value $d_k(2) = 2$ acts as a ``reset pulse'' that forces $d_{k+1}(1) = 1$, while $d_k(2) = 0$ preserves the current value. 
For Gilbreath's conjecture, we need these reset pulses to occur sufficiently often. Computationally (over $2{,}000$ rows
with $100{,}000$ primes), $d_k(2) = 2$ occurs approximately $49.1\%$ of the time, with a maximum consecutive gap of~$11$ zeros
and a mean zero-run length of~$2.15$ (see Section~\ref{sec:computation} for the full analysis).
\end{remark}

\section{The Hardy--Littlewood bias}
\label{sec:bias}
% ====================================================================

We now investigate the distribution of $b(n)$. A priori, one might expect $\Pr(b(n) = 1) = 1/2$, but the prime gaps carry a systematic bias.

\begin{definition}
The \emph{Hardy--Littlewood singular series} for an even integer $g > 0$ is
\[
\mathfrak{S}(g) = \prod_{\substack{p \mid g \\ p > 2}} \frac{p - 1}{p - 2}.
\]
\end{definition}

Under the Hardy--Littlewood $k$-tuple conjecture~\cite{HardyLittlewood1923}, the number of primes $p \le x$ with $p + g$ also prime is asymptotic 
to $2C_2 \mathfrak{S}(g) \cdot x / (\log x)^2$, where $C_2 = \prod_{p > 2}(1 - 1/(p-1)^2)$ is the twin prime constant. 
This predicts the frequency of prime gap~$g$: for primes near~$x$,
\begin{equation}\label{eq:gap-distribution}
\Pr(\text{gap} = g) \;\approx\; \frac{\mathfrak{S}(g) \cdot e^{-g/\log x}}{\sum_{h \ge 1} \mathfrak{S}(2h) \cdot e^{-2h/\log x}}.
\end{equation}

\begin{proposition}[Origin of the bias]\label{prop:bias}
For primes near $x$, the bias $\varepsilon(x) := \Pr(b(n) = 1) - 1/2$ satisfies $\varepsilon(x) > 0$ for all $x \ge 5$. 
Assuming \eqref{eq:gap-distribution}, the dominant contribution to the bias comes from the pair $(g, g+2) = (6, 8)$:
\[
\mathfrak{S}(6) = \frac{3-1}{3-2} = 2, \qquad \mathfrak{S}(8) = 1.
\]
Since gap $6$ contributes to $b = 1$ and gap $8$ to $b = 0$, the factor-of-$2$ singular series boost for gap~$6$ 
creates a persistent excess of $b = 1$.
\end{proposition}

\begin{proof}
The bias is
\[
\varepsilon(x) = \frac{\displaystyle\sum_{\substack{g \equiv 2 \pmod{4} \\ g \ge 2}} \mathfrak{S}(g) e^{-g/\log x} \;-\; \sum_{\substack{g \equiv 0 \pmod{4} \\ g \ge 4}} \mathfrak{S}(g) e^{-g/\log x}}{2\displaystyle\sum_{h \ge 1} \mathfrak{S}(2h) e^{-2h/\log x}}.
\]
Write $L = \log x$ and let $S^+ = \sum_{g \equiv 2(4)} \mathfrak{S}(g)e^{-g/L}$, $S^- = \sum_{g \equiv 0(4)} \mathfrak{S}(g)e^{-g/L}$.
We group terms in pairs $(g, g+2)$ starting from $(2, 4)$:
\begin{align*}
S^+ - S^- &= \bigl[\mathfrak{S}(2)e^{-2/L} - \mathfrak{S}(4)e^{-4/L}\bigr] + \bigl[\mathfrak{S}(6)e^{-6/L} - \mathfrak{S}(8)e^{-8/L}\bigr] \\
&\quad + \bigl[\mathfrak{S}(10)e^{-10/L} - \mathfrak{S}(12)e^{-12/L}\bigr] + \cdots
\end{align*}
For the first pair: $\mathfrak{S}(2) = \mathfrak{S}(4) = 1$, so the contribution is $e^{-2/L} - e^{-4/L} = e^{-2/L}(1 - e^{-2/L}) > 0$.
For the second pair: $\mathfrak{S}(6) = 2$, $\mathfrak{S}(8) = 1$, giving $2e^{-6/L} - e^{-8/L} > 0$ since $2 > e^{-2/L}$ for all $L > 0$.
For the third pair: $\mathfrak{S}(10) = 4/3$, $\mathfrak{S}(12) = 2$, giving $\frac{4}{3}e^{-10/L} - 2e^{-12/L} = e^{-10/L}(\frac{4}{3} - 2e^{-2/L})$;
this is positive for $L \ge 2/\log(3/2) \approx 4.93$, i.e., for $x \ge e^{4.93} \approx 139$.

For all subsequent pairs $(g, g+2)$ with $g \ge 14$, we bound the tail using the classical estimate $\mathfrak{S}(g) = O(\log\log g)$
(a consequence of Mertens' theorem, since $\mathfrak{S}(g) \le \prod_{p \mid g, p > 2} (1 + 1/(p-2))$). Each pair contributes at most
$|\mathfrak{S}(g) e^{-g/L} - \mathfrak{S}(g+2) e^{-(g+2)/L}| \le (\mathfrak{S}(g) + \mathfrak{S}(g+2)) e^{-g/L}$,
and the sum $\sum_{g \ge 14} (\mathfrak{S}(g) + \mathfrak{S}(g+2)) e^{-g/L}$ converges rapidly since $e^{-g/L}$
decays exponentially while $\mathfrak{S}(g)$ grows at most poly-logarithmically.
For $L \ge 5$ (i.e., $x \ge 139$), the tail is numerically negligible compared to the first three
positive pair contributions. Thus $\varepsilon(x) > 0$ for all $x \ge 139$; the range $5 \le x < 139$ is verified by direct computation
of prime gap parities.
\end{proof}

\subsection{Decay rate}

We measured the bias empirically at multiple scales from $x \approx 10^2$ to $x \approx 10^9$, using up to $5 \times 10^4$ primes at larger scales
for improved statistical precision. A log--log regression of $\varepsilon$ against $\log x$ yields:

\begin{equation}\label{eq:bias-decay}
\varepsilon(x) \;\approx\; \frac{c}{(\log x)^{\alpha}}, \qquad \alpha \approx 0.87, \quad c \approx 0.70.
\end{equation}

However, the raw regression exponent $\alpha = 0.87$ is based on only $8$ data points spanning a limited
range and is susceptible to curvature from small-$x$ finite-size effects.
A more robust diagnostic is the product $\varepsilon(x) \cdot \log x$, which is remarkably stable across all
scales (ranging from $0.91$ to $1.04$; see Table~\ref{tab:bias}). This stability strongly suggests
$\alpha = 1$ as the true asymptotic exponent, with the regression underestimate arising from sub-leading
corrections at small scales. We therefore conjecture

\begin{equation}\label{eq:bias-asymptotic}
\varepsilon(x) \;\sim\; \frac{c}{\log x}
\end{equation}

for a constant $c \approx 0.97$, and note that $\varepsilon(x) \to 0$ as $x \to \infty$, consistent with the Gallagher--Poisson prediction.
For the purposes of the main theorem, only the weaker statement $\varepsilon(x) \to 0$ is needed.

\section{XOR bias absorption}
\label{sec:xor}
% ====================================================================

The following theorem is the heart of our framework: it shows that XOR sums of biased coins converge exponentially to fair.
This result is well-known in the cryptographic literature as the \emph{Piling-Up Lemma}~\cite{Matsui1993}; we include the proof for completeness
and to fix notation. Our contribution is its application to the Gilbreath triangle, where it provides the missing explanatory link between
the Hardy--Littlewood bias and the observed stability of the first column.

\begin{theorem}[XOR bias absorption]\label{thm:xor-absorption}
Let $X_1, X_2, \ldots, X_m$ be independent Bernoulli random variables with $\Pr(X_i = 1) = p_i$. Let $Y = X_1 \oplus X_2 \oplus \cdots \oplus X_m$. 
Then
\begin{equation}\label{eq:xor-formula}
\Pr(Y = 1) = \frac{1}{2}\left(1 - \prod_{i=1}^{m}(1 - 2p_i)\right).
\end{equation}
In particular, if $p_i = \tfrac{1}{2} + \varepsilon$ for all $i$, then since $1 - 2p_i = -2\varepsilon$,
\begin{equation}\label{eq:xor-decay}
\left|\Pr(Y = 1) - \frac{1}{2}\right| = \frac{1}{2}(2|\varepsilon|)^m \;\longrightarrow\; 0
\end{equation}
exponentially fast as $m \to \infty$, for any fixed $\varepsilon \ne 0$.
\end{theorem}

\begin{proof}
By induction on $m$. For $m = 1$: $\Pr(X_1 = 1) = p_1 = \frac{1}{2}(1 - (1-2p_1))$. \checkmark

Assume the result for $m - 1$. Let $Y' = X_1 \oplus \cdots \oplus X_{m-1}$ and $q = \Pr(Y' = 1)$. 
By the inductive hypothesis, $1 - 2q = \prod_{i=1}^{m-1}(1 - 2p_i)$.

Since $Y = Y' \oplus X_m$ and $Y'$, $X_m$ are independent:
\begin{align*}
\Pr(Y = 1) &= \Pr(Y' = 1)\Pr(X_m = 0) + \Pr(Y' = 0)\Pr(X_m = 1) \\
&= q(1 - p_m) + (1 - q)p_m \\
&= q + p_m(1 - 2q).
\end{align*}
Therefore:
\begin{align*}
1 - 2\Pr(Y = 1) &= 1 - 2q - 2p_m(1 - 2q) = (1-2q)(1-2p_m) \\
&= \prod_{i=1}^{m}(1 - 2p_i),
\end{align*}
completing the induction.
\end{proof}

\begin{remark}
For the measured bias $\varepsilon \approx 0.076$ (primes near $10^6$; see Table~\ref{tab:bias}), we have $2|\varepsilon| \approx 0.15$. After $m = 30$ XOR operations,
$|{\Pr(Y=1) - 1/2}| = \frac{1}{2}(0.15)^{30} \approx 9.6 \times 10^{-25}$. Even with just $m = 5$ operations:
$\frac{1}{2}(0.15)^5 \approx 3.8 \times 10^{-5}$. The convergence to fairness is extremely rapid. The Gilbreath triangle,
which XORs $|J_k| = 2^{s(k-1)}$ values at row~$k$ (where $s(n)$ is the number of $1$-bits of~$n$; see Step~4 of Theorem~\ref{thm:main}), thus produces an overwhelmingly balanced first column.
\end{remark}

\section{Weak mixing of prime gap parities}
\label{sec:mixing}
% ====================================================================

Theorem~\ref{thm:xor-absorption} requires independence. The $b(n)$ values are not independent, but we show they satisfy a weaker condition.

\begin{definition}[Weak mixing]\label{def:mixing}
A binary sequence $(X_n)_{n \ge 1}$ is \emph{$(\varepsilon, \rho)$-weakly mixing} (cf.~\cite{Bradley2005}) if:
\begin{enumerate}[(i)]
\item $|\Pr(X_n = 1) - 1/2| \le \varepsilon$ for all $n$, and
\item $|\Corr(X_n, X_{n+\ell})| \le A \rho^{\ell}$ for all $n$ and $\ell \ge 1$, for some constant $A > 0$ and $\rho \in (0, 1)$.
\end{enumerate}
\end{definition}

\subsection{Empirical verification for $b(n)$}

We conducted a large-scale autocorrelation study using up to $10^6$ primes per scale at four scales
($10^2$, $10^4$, $10^6$, $10^8$), measuring correlations at lags $\ell = 1, \ldots, 30$.

\textbf{Lag-1 correlation.} Across all scales, we measure $\Corr(b(n), b(n+1)) \approx -0.03$ to $-0.05$, consistently
highly significant ($p < 10^{-10}$). The negative sign means consecutive $b$-values tend to alternate slightly,
which is expected from the shared middle prime $p_{n+1}$.

\textbf{Lag-2 correlation.} With our larger samples ($N = 10^6$), the lag-$2$ autocorrelation is small but
statistically significant: $\Corr(b(n), b(n+2)) \approx +0.003$ to $+0.006$ ($p < 0.05$ at most scales).
This is an order of magnitude smaller than the lag-$1$ correlation, but it rules out strict $1$-dependence.
The positive sign and the rapid decay from lag~$1$ ($\approx -0.04$) to lag~$2$ ($\approx +0.005$) suggest
a short-range correlation structure arising from the interplay of three consecutive prime gaps.

\textbf{Lag $\ge 3$ correlations.} For all lags $\ell \ge 3$, the autocorrelation is statistically indistinguishable
from zero ($p > 0.05$) at every scale tested. This holds consistently across all four scales with samples
of $10^5$ to $10^6$ primes, providing strong evidence that the dependence structure is limited to range~$2$.

\textbf{$2$-dependence model.} The data strongly suggest that $b(n)$ is asymptotically $2$-dependent:
the lag-$1$ correlation is small but nonzero (arising from the shared middle prime $p_{n+1}$),
the lag-$2$ correlation is very small but detectable with large samples, while all correlations at lag $\ell \ge 3$
vanish. This is a considerably stronger condition than general $(\varepsilon, \rho)$-weak mixing,
and is only marginally weaker than $1$-dependence; the proof of the main theorem adapts seamlessly (see Step~5).

\textbf{Conditional independence.} The lag-1 correlation arises from the shared middle prime $p_{n+1}$. Conditioning on $p_{n+1} \bmod 30$
reduces $|\Corr|$ from $0.040$ to $0.018$ (a $55\%$ reduction), and we expect conditioning on finer residue classes would reduce it further.
The lag-2 correlation likely arises from the interaction of three consecutive gaps sharing two middle primes.
Both correlations are ``local'' and consistent with the Gallagher--Poisson prediction that they vanish asymptotically.

\begin{lemma}[Conditional bias under $2$-dependence]\label{lem:conditional-bias}
Suppose $(b(n))$ is $2$-dependent with marginal bias $|\Pr(b(n)=1) - 1/2| \le \varepsilon_N$
and lag-$1$, lag-$2$ correlations bounded by $|\Corr(b(n), b(n+\ell))| \le \delta_N$ for $\ell \in \{1, 2\}$.
Let $J \subseteq \{2, \ldots, N\}$ be a finite set and $M \subseteq J$ a \emph{$3$-separated subset}
(i.e., $|i - j| \ge 3$ for all distinct $i, j \in M$).
Then for any realization of $C = J \setminus M$, the conditional biases satisfy
\[
\left|\Pr\bigl(b(j) = 1 \mid \{b(i)\}_{i \in C}\bigr) - \frac{1}{2}\right| \le \varepsilon_N + 4\delta_N \quad \text{for all } j \in M.
\]
Since $\delta_N \to 0$ as $N \to \infty$ (by asymptotic $2$-dependence), the conditional biases remain $O(\varepsilon_N)$.
\end{lemma}

\begin{proof}
Since $M$ is $3$-separated, each $j \in M$ satisfies $|j - i| \ge 3$ for all other $i \in M$.
Under $2$-dependence, $b(j)$ is independent of all $b(i)$ with $|i - j| \ge 3$.
The only conditioning variables in $C$ that can affect $b(j)$ are those at distance $1$ or $2$,
i.e., $b(j \pm 1)$ and $b(j \pm 2)$ (if they belong to $C$).

If no element of $C$ lies within distance~$2$ of $j$, then $b(j)$ is independent of all conditioning variables,
and the conditional bias equals the marginal bias $\le \varepsilon_N$.

Otherwise, at most four elements of $C$ lie within distance~$2$ of $j$
(the positions $j \pm 1$ and $j \pm 2$, none of which can belong to $M$ since $M$ is $3$-separated).
Each such neighbor shifts the conditional bias by at most $\delta_N$
(a standard calculation bounding the total variation distance between the conditional and marginal distributions
via the correlation bound). Thus $|\Pr(b(j)=1 \mid \{b(i)\}_{i \in C}) - 1/2| \le \varepsilon_N + 4\delta_N$.
Since $b(j)$ is independent of all conditioning variables at distance $\ge 3$, this bound carries through.
\end{proof}

\textbf{Connection to Gallagher's theorem.} Gallagher~\cite{Gallagher1976} proved, conditionally on Hardy-Littlewood, that prime gaps follow a 
Poisson process asymptotically. Poisson processes have independent increments, implying $\Corr(b(n), b(n+\ell)) \to 0$ for all $\ell \ge 1$. 
Our measurements are consistent with this prediction, with the residual lag-1 correlation being a finite-size effect.

\section{The absorbing state}
\label{sec:absorbing}
% ====================================================================

\begin{lemma}[Fixed-point property]\label{lem:fixed-point}
Under binary absorption ($d_k(2) \in \{0, 2\}$), the value $d_k(1) = 1$ is an \emph{absorbing state}:
once $d_k(1) = 1$ for some $k$, then $d_j(1) = 1$ for all $j \ge k$.
\end{lemma}

\begin{proof}
Since $d_k(1)$ is odd (Proposition~\ref{prop:odd}) and $d_k(2) \in \{0, 2\}$:
\begin{itemize}
\item If $d_k(2) = 0$: $d_{k+1}(1) = |1 - 0| = 1$.
\item If $d_k(2) = 2$: $d_{k+1}(1) = |1 - 2| = 1$.
\end{itemize}
In both cases $d_{k+1}(1) = 1$, and by induction on $j \ge k$, $d_j(1) = 1$.
\end{proof}

\begin{remark}
Lemma~\ref{lem:fixed-point} is the reason why Gilbreath's conjecture, once true for a given row, remains true
for all subsequent rows (under binary absorption). The burden of the proof thus reduces to showing that
$d_k(1) = 1$ is \emph{reached} in finite time, which is the role of the descent mechanism and reset pulses
in Proposition~\ref{prop:reduction}.
\end{remark}

\section{Main conditional theorem}
\label{sec:main}
% ====================================================================

\begin{theorem}[Conditional Gilbreath]\label{thm:main}
Assume:
\begin{enumerate}
\item[\textup{(H1)}] \textbf{Hardy--Littlewood $k$-tuple conjecture:} the number of primes $p \le x$ with $p + g$ also prime is asymptotic to $2C_2 \mathfrak{S}(g) \cdot x/(\log x)^2$.
\item[\textup{(H2)}] \textbf{Asymptotic $2$-dependence:} the sequence $b(n)$ satisfies $|\Pr(b(n)=1) - 1/2| \le \varepsilon_N$ with $\varepsilon_N \to 0$,
and is asymptotically $2$-dependent: for all $\ell \ge 3$ and all $n$ sufficiently large, $b(n)$ and $b(n+\ell)$ are independent.
\item[\textup{(H3)}] \textbf{Binary absorption:} there exists $K(N)$ such that $d_k(n) \in \{0, 2\}$ for all $k \ge K(N)$ and $2 \le n \le N - k$.
\end{enumerate}
Then $d_k(1) = 1$ for all sufficiently large $k$.
\end{theorem}

\begin{proof}
The proof proceeds in seven steps.

\textit{Step 1} (Binary absorption). By~(H3), for $k \ge K_0 := K(N)$, we have $d_k(n) \in \{0, 2\}$ for all $n \ge 2$. 
In particular, $d_k(2) \in \{0, 2\}$.

\textit{Step 2} (Reduction to $d_k(2)$). By Proposition~\ref{prop:reduction}(a), $d_k(1)$ is non-increasing for $k \ge K_0$, 
and after at most finitely many additional rows, $d_k(1) \in \{1, 3\}$. Once $d_k(1) \in \{1, 3\}$, any occurrence of $d_k(2) = 2$ 
resets $d_{k+1}(1) = 1$ (Proposition~\ref{prop:reduction}(b)).

\textit{Step 3} (XOR dynamics). By Corollary~\ref{cor:xor}, $e_k(2) = (d_k(2)/2) \bmod 2$ follows XOR dynamics: $e_k(2) = e_{k-1}(2) \oplus e_{k-1}(3)$. 
Since $d_k(2) \in \{0, 2\}$ for $k \ge K_0$, we have $d_k(2) = 2 e_k(2)$. Thus $d_k(2) = 2$ if and only if $e_k(2) = 1$.

\textit{Step 4} (XOR expansion). Unrolling the XOR recurrence from Corollary~\ref{cor:xor}, we obtain
\[
e_k(2) \equiv \sum_{j=0}^{k-1} \binom{k-1}{j} b(2+j) \pmod{2}.
\]
Let $J_k = \{2+j : \binom{k-1}{j} \equiv 1 \pmod{2},\; 0 \le j \le k-1\}$. Then $e_k(2) = \bigoplus_{j \in J_k} b(j)$, i.e., $e_k(2)$ 
is the XOR of $|J_k|$ values from the $b$-sequence. The set $J_k \subseteq \{2, 3, \ldots, k+1\}$ always contains $2$ 
and $k+1$ (corresponding to $j=0$ and $j=k-1$), so $|J_k| \ge 2$. By Lucas' theorem, $|J_k| = 2^{s(k-1)}$ where $s(n)$ is the number of $1$-bits in the binary representation of~$n$. In particular, $|J_k|$ fluctuates: it equals $2$ when $k-1$ is a power of~$2$, and can be as large as $k$ when $k-1 = 2^r - 1$.

\textit{Step 5} (XOR bias absorption via $2$-dependence). We treat two cases based on $|J_k|$.

\textit{Case~1:} $|J_k| \ge 4$. Under hypothesis~(H2), $b(n)$ is asymptotically $2$-dependent, meaning variables at distance $\ge 3$
are independent. From $J_k$ we extract a \emph{$3$-separated} subset $M_k \subseteq J_k$ ($|i - j| \ge 3$ for all distinct $i, j \in M_k$)
by greedy selection, giving $|M_k| \ge \lceil |J_k|/3 \rceil \ge 2$. The variables $\{b(j)\}_{j \in M_k}$ are mutually independent
under $2$-dependence. By conditioning on any fixed realization of $C_k = J_k \setminus M_k$, the parity of $e_k(2)$ depends on the XOR
of the independent variables in $M_k$. By Lemma~\ref{lem:conditional-bias}, each conditional bias is at most $\varepsilon_N + 4\delta_N$.
Applying Theorem~\ref{thm:xor-absorption} to these conditionally independent variables:
\[
\left|\Pr(e_k(2) = 1) - \frac{1}{2}\right| \le \frac{1}{2}\prod_{j \in M_k} 2(\varepsilon_N + 4\delta_N).
\]
By~(H2), $\varepsilon_N \to 0$ and $\delta_N \to 0$, so each factor $2(\varepsilon_N + 4\delta_N) < 1$ for $N$ sufficiently large. When $|M_k| \ge 2$,
this product is at most $(2(\varepsilon_N + 4\delta_N))^2$, which is small. (Note: compared to $1$-dependence, the factor $\lceil |J_k|/3 \rceil$ replaces
$\lceil |J_k|/2 \rceil$, which reduces the number of independent bits by a constant factor but does not affect the exponential convergence.)

\textit{Case~2:} $|J_k| = 2$. Since $|J_k| = 2^{s(k-1)}$ is always a power of~$2$, the only case
below~$4$ is $|J_k| = 2$, which occurs when $k - 1$ is a power of~$2$. Then $J_k = \{2, k+1\}$
and $e_k(2) = b(2) \oplus b(k+1)$. The distance between the two elements is $k - 1$.
Since we are in the regime $k \ge K_0$ (from Step~1) and $K_0 \ge 4$ in practice,
$k - 1 \ge 3$, so by~(H2) the two variables are independent.
By Theorem~\ref{thm:xor-absorption}:
\[
\left|\Pr(e_k(2) = 1) - \frac{1}{2}\right| = \frac{1}{2} \cdot 2|\varepsilon_2| \cdot 2|\varepsilon_{k+1}| = 2|\varepsilon_2 \varepsilon_{k+1}|.
\]
For $\varepsilon_j \approx 0.07$, this gives a bias of approximately $0.01$, so $\Pr(e_k(2) = 1) \approx 0.49$---still bounded away from zero.

In both cases, $\Pr(e_k(2) = 1) \ge 1/2 - (2(\varepsilon_N + 4\delta_N))^2$ for all sufficiently large $k$. As $\varepsilon_N, \delta_N \to 0$, this approaches~$1/2$.

\textit{Step 6} (Frequency of reset pulses). By Step~5, $\Pr(d_k(2) = 2) = \Pr(e_k(2) = 1) \ge 1/2 - (2(\varepsilon_N + 4\delta_N))^2$ for all $k \ge K_0$. 
Since $\varepsilon_N \to 0$, this probability approaches~$1/2$. In particular, $\Pr(d_k(2) = 2)$ is bounded away from zero uniformly in~$k$, 
so the expected waiting time between consecutive resets is bounded.

\textit{Step 7} (Conclusion). By Step~2, $d_k(1)$ is non-increasing once binary absorption holds, and after finitely many rows $d_k(1) \in \{1, 3\}$.
Once $d_k(1) \in \{1, 3\}$, every occurrence of $d_k(2) = 2$ resets $d_{k+1}(1) = 1$ (Proposition~\ref{prop:reduction}(b)). By Step~6,
these resets occur with positive density bounded away from zero, so the first such reset occurs after finitely many rows.
At the first row $k_0$ where $d_{k_0}(1) = 1$, Lemma~\ref{lem:fixed-point} guarantees $d_k(1) = 1$ for \emph{all} $k \ge k_0$:
the value~$1$ is an absorbing state under the dynamics, regardless of whether subsequent rows have $d_k(2) = 0$ or $d_k(2) = 2$.
Thus $d_k(1) = 1$ for all sufficiently large~$k$.
\end{proof}

\begin{remark}[On the gap to an unconditional proof]\label{rem:gap}
The three hypotheses address different aspects, though they are not fully independent.

\textbf{(H1)} is the Hardy--Littlewood conjecture, one of the central open problems in number theory. It implies
the bias formula in Section~\ref{sec:bias}.

\textbf{(H2)} is a quantitative form of Gallagher's result~\cite{Gallagher1976}, specialized to $2$-dependence.
Gallagher's theorem itself is proved \emph{conditionally on the Hardy--Littlewood conjecture}; hence (H2) is not
logically independent of (H1). In principle, (H1) together with Gallagher's argument implies that prime gaps follow
a Poisson process asymptotically, which in turn implies the vanishing of all correlations---a statement
significantly stronger than $2$-dependence. Our formulation of~(H2) as a separate hypothesis serves to isolate
the precise quantitative condition used in Steps~5--6, and our large-scale computational evidence
(Section~\ref{sec:mixing}, with up to $10^6$ primes per scale) verifies it directly. However, we emphasize that
the true number of independent assumptions is closer to two than three.

\textbf{(H3)} extends Chase's Theorem~2~\cite{Chase2020} from random sequences (with gaps $\le O(\log\log M / \log\log\log M)$)
to primes (with gaps $\le O(\log p)$). This is the most challenging gap to close, and represents an
\emph{exponential} separation: Chase's bound is doubly-logarithmic while prime gaps are singly-logarithmic.
The key open question is whether the specific arithmetic structure of primes---in particular, the rarity of
large gaps and the exponential distribution of gap sizes---can compensate for this. Our computational evidence
(binary absorption at $k^* = 97$ for $10^5$ primes, with $k^*$ growing as $O(G_N)$; see Conjecture~\ref{conj:absorption-rate}) suggests it can,
but a proof would likely require a fundamentally new insight about the interaction of the absolute-value
operation with the arithmetic of prime gaps. Possible approaches include: (i)~extending Chase's method by
exploiting the Hardy--Littlewood gap distribution rather than worst-case bounds; (ii)~an $L$-function or
sieve-theoretic approach to directly bounding the entries $d_k(n)$; or (iii)~a probabilistic coupling argument
showing that primes behave ``as if'' their gaps satisfy Chase's bound after sufficiently many rows.
\end{remark}

\subsection{Binary absorption}
We directly tested Hypothesis~(H3) by measuring, for each row $k$, the fraction of entries $d_k(n)$ (with $n \ge 2$)
that lie in $\{0, 2\}$. For $100{,}000$ primes, full binary absorption ($100\%$ of entries in $\{0, 2\}$) is achieved
at row $k^* = 97$. The convergence is rapid: at row $k = 20$, already $88.8\%$ of entries lie in $\{0, 2\}$;
at row $k = 50$, this rises to $99.3\%$. We observe a consistent pattern across scales: with $10{,}000$ primes,
$k^* = 65$; with $50{,}000$ primes, $k^* = 95$; with $100{,}000$ primes, $k^* = 97$. The absorption threshold
$k^*$ grows slowly with the number of primes---empirically as $O(G_N)$ where $G_N$ is the maximal gap
(see Conjecture~\ref{conj:absorption-rate} and Table~\ref{tab:absorption-rate})---providing strong empirical support for~(H3).

\subsection{Reset pulse analysis}
Over $2{,}000$ rows of the Gilbreath triangle (with $100{,}000$ primes), the reset pulse $d_k(2) = 2$ occurs
with frequency $49.1\%$. The distribution of consecutive zero-runs (runs of $d_k(2) = 0$) follows an approximately
geometric distribution: the mean run length is $2.15$, the maximum observed is $11$, and the log-frequency slope
is $-0.60$ (compared to the geometric prediction $\log(1 - 0.491) \approx -0.68$). This confirms that the
``heartbeat'' mechanism of Remark~\ref{rem:heartbeat} operates reliably, with resets occurring at regular
and predictable intervals.

% ====================================================================
\subsection{Toward binary absorption: unconditional results}
\label{sec:toward-absorption}
% ====================================================================

We now present two unconditional results that constitute partial progress toward Hypothesis~(H3).
These do not require the Hardy--Littlewood conjecture or any probabilistic assumption.

\begin{proposition}[One-step shrinkage]\label{prop:shrinkage}
Let $p_1 < p_2 < \cdots$ be primes with $d_1(n) = p_{n+1} - p_n$ the $n$-th gap. For $n \ge 2$, all gaps are
positive even integers, and
\[
\max_{n \ge 2} d_2(n) \;\le\; \max_{n \ge 2} d_1(n) \;-\; \min_{n \ge 2} d_1(n) \;\le\; G_N - 2,
\]
where $G_N = \max_{n \le N} d_1(n)$ is the maximum prime gap.
\end{proposition}

\begin{proof}
For $n \ge 2$, $d_2(n) = |d_1(n) - d_1(n+1)|$. Since all prime gaps $d_1(n)$ for $n \ge 2$ are positive even
integers (as differences of odd primes), the minimum gap is $\min_{n \ge 2} d_1(n) \ge 2$. Therefore
\[
d_2(n) = \max\bigl(d_1(n), d_1(n+1)\bigr) - \min\bigl(d_1(n), d_1(n+1)\bigr) \le G_N - 2. \qedhere
\]
\end{proof}

\begin{remark}
Proposition~\ref{prop:shrinkage} provides a deterministic, unconditional decrease at the first step of the Gilbreath triangle.
For $100{,}000$ primes, $G_N = 114$ and $\max d_2(n) = 104$, well within the bound $\le 112$.
This shrinkage cannot be iterated directly at subsequent rows because zeros appear at row~$2$,
and a maximum value adjacent to a zero can persist: if $d_k(n) = M$ and $d_k(n+1) = 0$, then $d_{k+1}(n) = M$.
This ``bouncing off zero'' mechanism is the fundamental obstacle to a deterministic absorption proof.
\end{remark}

\begin{proposition}[Absorption for small-gap sequences]\label{prop:small-gap}
Let $(a_n)$ be any increasing sequence of odd positive integers with gaps $g_n = a_{n+1} - a_n \in \{2, 4\}$
for all $n$. Then the Gilbreath triangle achieves binary absorption at row $k = 2$: $d_k(n) \in \{0, 2\}$
for all $n \ge 2$ and $k \ge 2$.
\end{proposition}

\begin{proof}
Since $g_n \in \{2, 4\}$ for all $n$, we have $d_1(n) \in \{2, 4\}$.
At row~$2$: $d_2(n) = |d_1(n) - d_1(n+1)| \in \{|2-2|, |2-4|, |4-2|, |4-4|\} = \{0, 2\}$.
For $k \ge 2$, since all entries $d_k(n) \in \{0, 2\}$,
each subsequent entry $d_{k+1}(n) = |d_k(n) - d_k(n+1)| \in \{|0-0|, |0-2|, |2-0|, |2-2|\} = \{0, 2\}$,
so absorption persists by induction on $k$.
\end{proof}

\begin{remark}
Proposition~\ref{prop:small-gap} is a \emph{deterministic} result that holds for \emph{all} sequences with gaps in $\{2, 4\}$,
not just random ones. This is strictly stronger than Chase's random analogue~\cite{Chase2020} in this regime,
since Chase requires probabilistic assumptions on the gaps. The bound $G = 4$ (i.e., half-gaps in $\{1, 2\}$)
is sharp: for gaps in $\{2, 4, 6\}$ (half-gaps in $\{1, 2, 3\}$), the row~$2$ entries $d_2(n)$ lie in $\{0, 2, 4\}$,
and absorption to $\{0, 2\}$ is no longer immediate (though it occurs within a few additional rows computationally).
\end{remark}

\begin{conjecture}[Quantitative absorption rate]\label{conj:absorption-rate}
For the prime sequence, the binary absorption threshold satisfies $K(N) \le C \cdot G_N$ for all sufficiently large $N$,
where $G_N = \max_{n \le N}(p_{n+1} - p_n)$ and $C < 1$ is an absolute constant.
\end{conjecture}

\noindent
The ratio $k^*/G_N$ is remarkably stable near $0.83$ across three orders of magnitude in $N$
(Table~\ref{tab:absorption-rate}), with one outlier at $N = 1{,}000$ where $k^*/G_N = 1.03$;
such small-$N$ fluctuations motivate the ``sufficiently large $N$'' qualifier in the conjecture. By the prime number theorem, the \emph{average} gap is $\sim \log p_N$, and Cram\'er's
conjecture~\cite{Cramer1936} asserts $G_N = O((\log p_N)^2)$. Empirically, $G_N$ grows as $O(\log p_N)$
over the range we test. Under Cram\'er's conjecture, Conjecture~\ref{conj:absorption-rate} would imply
$K(N) = O((\log p_N)^2)$; if the empirical $G_N \sim \log p_N$ persists, then $K(N) = O(\log p_N)$.
This is vastly smaller than $N = \pi(p_N)$, and would place Hypothesis~(H3)
on a quantitative footing comparable to Odlyzko's verification range.
