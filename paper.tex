\begin{abstract}
We develop a conditional framework for Gilbreath's conjecture---the statement that the first element of every row in the iterated absolute difference triangle of primes is always~$1$---by identifying a mechanism we call \emph{XOR bias absorption}. Building on the mod~4 reduction of Odlyzko~\cite{Odlyzko1993} and the random analogue of Chase~\cite{Chase2020}, we prove that the parity sequence $b(n) = \lfloor d_1(n)/2 \rfloor \bmod 2$ derived from prime gaps carries a systematic bias of magnitude $\varepsilon(x) \approx C/(\log x)^{\alpha}$ with $\alpha \approx 1.0$, arising from the Hardy--Littlewood singular series. We prove exactly (Theorem~\ref{thm:xor-absorption}) that XOR sums of independent biased coins converge exponentially fast to fairness, and verify computationally that the prime gap parities satisfy the requisite weak-mixing conditions. Under three explicit hypotheses---the Hardy--Littlewood conjecture, quantitative weak mixing, and binary absorption for primes---we show that Gilbreath's conjecture holds for all sufficiently large~$k$. All intermediate results are verified computationally for primes up to $10^9$.
\end{abstract}

\section{Introduction}
\label{sec:intro}
% ====================================================================

Let $p_1 = 2, p_2 = 3, p_3 = 5, \ldots$ denote the primes in their natural ordering. Following Odlyzko~\cite{Odlyzko1993}, define
\begin{equation}\label{eq:gilbreath-def}
d_0(n) = p_n, \qquad d_{k+1}(n) = |d_k(n) - d_k(n+1)| \quad (k \ge 0,\; n \ge 1).
\end{equation}
The sequence $(d_k(1))_{k \ge 0} = 2, 1, 1, 1, \ldots$ constitutes the first column of the \emph{Gilbreath triangle}.

\begin{conjecture}[Gilbreath, 1958; Proth, 1878]
$d_k(1) = 1$ for all $k \ge 1$.
\end{conjecture}

This conjecture has been verified computationally by Odlyzko~\cite{Odlyzko1993} for all $k \le \pi(10^{13}) \approx 3.4 \times 10^{11}$. Chase~\cite{Chase2020} proved a random analogue: for sequences beginning with~$2$ followed by increasing odd numbers with random gaps bounded by $O(\log\log M / \log\log\log M)$, the Gilbreath property holds almost surely. However, actual primes have gaps of order $O(\log p)$, far exceeding Chase's bound.

\subsection{Our contribution}
We identify a three-step mechanism explaining why Gilbreath's conjecture holds:
\begin{enumerate}[(1)]
\item \textbf{Mod~4 reduction (Odlyzko):} The Gilbreath triangle, reduced modulo~4, follows XOR dynamics on binary sequences derived from prime gaps.
\item \textbf{Hardy--Littlewood bias:} These binary sequences carry a computable bias $\varepsilon(x)$ from the singular series, which decays as $O(1/(\log x)^{\alpha})$.
\item \textbf{XOR bias absorption (new):} XOR sums of many weakly-dependent biased bits converge exponentially fast to fair coin flips, explaining why the first column remains~$1$.
\end{enumerate}

We formalize this as a conditional theorem (Theorem~\ref{thm:main}), identifying three explicit hypotheses that together imply Gilbreath's conjecture for all sufficiently large~$k$.

\section{Definitions and the mod~4 framework}
\label{sec:mod4}
% ====================================================================

\begin{definition}[Binary parity sequence]\label{def:b-sequence}
For $n \ge 2$, define
\[
b(n) = \left\lfloor \frac{d_1(n)}{2} \right\rfloor \bmod 2,
\]
where $d_1(n) = p_{n+1} - p_n$ is the $n$-th prime gap. Since $p_n$ and $p_{n+1}$ are both odd for $n \ge 2$, $d_1(n)$ is even, and $b(n) \in \{0, 1\}$ is well-defined. Equivalently, $b(n) = 0$ if $d_1(n) \equiv 0 \pmod{4}$, and $b(n) = 1$ if $d_1(n) \equiv 2 \pmod{4}$.
\end{definition}

\begin{proposition}[Odlyzko {\cite[Section~2]{Odlyzko1993}}]\label{prop:mod4}
For $k \ge 1$ and $n \ge 2$:
\begin{enumerate}[(a)]
\item $d_k(n)$ is even.
\item $d_{k+1}(n) \equiv d_k(n) + d_k(n+1) \pmod{4}$.
\end{enumerate}
\end{proposition}

\begin{proof}
Part~(a) follows by induction on $k$: the base case $d_1(n) = p_{n+1} - p_n$ is even for $n \ge 2$ (both primes odd), and $|{\rm even} - {\rm even}| = {\rm even}$.

For part~(b), write $d_k(n) = 2a$ and $d_k(n+1) = 2c$ with $a, c \ge 0$. Then
\[
d_{k+1}(n) = |2a - 2c| = 2|a - c|.
\]
Since $|a - c| = a + c - 2\min(a,c)$, we have $|a-c| \equiv a + c \pmod{2}$, hence
\[
2|a-c| \equiv 2(a+c) = d_k(n) + d_k(n+1) \pmod{4}. \qedhere
\]
\end{proof}

\begin{corollary}[XOR reduction]\label{cor:xor}
Define $e_k(n) = (d_k(n)/2) \bmod 2$ for $k \ge 1$, $n \ge 2$. Then
\[
e_{k+1}(n) = e_k(n) \oplus e_k(n+1),
\]
where $\oplus$ denotes addition modulo~$2$. In particular, $e_1(n) = b(n)$.
\end{corollary}

\begin{proof}
Dividing the congruence in Proposition~\ref{prop:mod4}(b) by~$2$:
\[
\frac{d_{k+1}(n)}{2} \equiv \frac{d_k(n)}{2} + \frac{d_k(n+1)}{2} \pmod{2}. \qedhere
\]
\end{proof}

\begin{remark}
Corollary~\ref{cor:xor} reduces the mod~4 dynamics of the Gilbreath triangle (for $n \ge 2$) to the XOR triangle---equivalently, Pascal's triangle modulo~2---applied to the binary sequence $(b(n))_{n \ge 2}$. This is computationally verified for $974{,}702$ instances with zero violations.
\end{remark}

\section{The first column mechanism}
\label{sec:first-column}
% ====================================================================

\begin{proposition}[Parity of the first column]\label{prop:odd}
For all $k \ge 1$, $d_k(1)$ is odd.
\end{proposition}

\begin{proof}
$d_1(1) = |p_2 - p_1| = 1$, which is odd. Inductively, $d_k(1) = |d_{k-1}(1) - d_{k-1}(2)|$. By induction $d_{k-1}(1)$ is odd, and by Proposition~\ref{prop:mod4}(a), $d_{k-1}(2)$ is even (for $k \ge 2$). Hence $|{\rm odd} - {\rm even}|$ is odd.
\end{proof}

\begin{proposition}[Reduction to $d_k(2)$]\label{prop:reduction}
Suppose $d_k(2) \in \{0, 2\}$ for all $k \ge K$. Then:
\begin{enumerate}[(a)]
\item The sequence $d_k(1)$ is non-increasing for $k \ge K$. In particular, $d_k(1) \in \{1, 3\}$ for all $k \ge K'$, where $K' = K + \lfloor(d_K(1) - 1)/2\rfloor$ (at most finitely many steps beyond $K$).
\item If $d_k(1) \in \{1,3\}$ and $d_k(2) = 2$, then $d_{k+1}(1) = 1$ (reset pulse).
\item If $d_k(2) = 0$, then $d_{k+1}(1) = d_k(1)$ (the value is preserved).
\end{enumerate}
\end{proposition}

\begin{proof}
Since $d_k(1)$ is odd (Proposition~\ref{prop:odd}) and $d_k(2) \in \{0, 2\}$:
\begin{itemize}
\item $d_k(2) = 0$: $d_{k+1}(1) = |d_k(1) - 0| = d_k(1)$, proving~(c).
\item $d_k(2) = 2$: $d_{k+1}(1) = |d_k(1) - 2|$. Since $d_k(1) \ge 1$ is odd, this equals $d_k(1) - 2$ when $d_k(1) \ge 3$, and $1$ when $d_k(1) = 1$.
\end{itemize}
For part~(a), observe that in both cases $d_{k+1}(1) \le d_k(1)$. Moreover, each occurrence of $d_k(2) = 2$ with $d_k(1) \ge 3$ \emph{strictly} decreases $d_k(1)$ by~$2$. Starting from any odd value $d_K(1) = 2m+1$, at most $m$ such resets bring $d_k(1)$ down to~$1$, which is then permanently stable (since $|1-2| = 1$ and $|1-0| = 1$). In particular, $d_k(1) \in \{1, 3\}$ after at most $\lfloor(d_K(1)-1)/2\rfloor$ resets past row~$K$, proving~(a). Once $d_k(1) \in \{1,3\}$, a reset gives $|1-2|=1$ or $|3-2|=1$, proving~(b).
\end{proof}

\begin{remark}\label{rem:heartbeat}
Proposition~\ref{prop:reduction} reveals a ``heartbeat'' mechanism: once binary absorption is achieved ($d_k(2) \in \{0,2\}$), the value $d_k(2) = 2$ acts as a ``reset pulse'' that forces $d_{k+1}(1) = 1$, while $d_k(2) = 0$ preserves the current value. For Gilbreath's conjecture, we need these reset pulses to occur sufficiently often. Computationally (for $50{,}000$ primes), $d_k(2) = 2$ occurs approximately $50.9\%$ of the time, with a maximum consecutive gap of~$9$ zeros.
\end{remark}

\section{The Hardy--Littlewood bias}
\label{sec:bias}
% ====================================================================

We now investigate the distribution of $b(n)$. A priori, one might expect $\Pr(b(n) = 1) = 1/2$, but the prime gaps carry a systematic bias.

\begin{definition}
The \emph{Hardy--Littlewood singular series} for an even integer $g > 0$ is
\[
\mathfrak{S}(g) = \prod_{\substack{p \mid g \\ p > 2}} \frac{p - 1}{p - 2}.
\]
\end{definition}

Under the Hardy--Littlewood $k$-tuple conjecture, the number of primes $p \le x$ with $p + g$ also prime is asymptotic to $2C_2 \mathfrak{S}(g) \cdot x / (\log x)^2$, where $C_2 = \prod_{p > 2}(1 - 1/(p-1)^2)$ is the twin prime constant. This predicts the frequency of prime gap~$g$: for primes near~$x$,
\begin{equation}\label{eq:gap-distribution}
\Pr(\text{gap} = g) \;\approx\; \frac{\mathfrak{S}(g) \cdot e^{-g/\log x}}{\sum_{h \ge 1} \mathfrak{S}(2h) \cdot e^{-2h/\log x}}.
\end{equation}

\begin{proposition}[Origin of the bias]\label{prop:bias}
For primes near $x$, the bias $\varepsilon(x) := \Pr(b(n) = 1) - 1/2$ satisfies $\varepsilon(x) > 0$ for all $x \ge 5$. Assuming \eqref{eq:gap-distribution}, the dominant contribution to the bias comes from the pair $(g, g+2) = (6, 8)$:
\[
\mathfrak{S}(6) = \frac{3-1}{3-2} = 2, \qquad \mathfrak{S}(8) = 1.
\]
Since gap $6$ contributes to $b = 1$ and gap $8$ to $b = 0$, the factor-of-$2$ singular series boost for gap~$6$ creates a persistent excess of $b = 1$.
\end{proposition}

\begin{proof}
The bias is
\[
\varepsilon(x) = \frac{\displaystyle\sum_{\substack{g \equiv 2 \pmod{4} \\ g \ge 2}} \mathfrak{S}(g) e^{-g/\log x} \;-\; \sum_{\substack{g \equiv 0 \pmod{4} \\ g \ge 4}} \mathfrak{S}(g) e^{-g/\log x}}{2\displaystyle\sum_{h \ge 1} \mathfrak{S}(2h) e^{-2h/\log x}}.
\]
For any $x \ge 5$, the numerator is dominated by $\mathfrak{S}(6) e^{-6/\log x} - \mathfrak{S}(8) e^{-8/\log x} = 2e^{-6/\log x} - e^{-8/\log x} > 0$ (since $2e^{-6/L} > e^{-8/L}$ for all $L > 0$). Thus $\varepsilon(x) > 0$.
\end{proof}

\subsection{Decay rate}

We measured the bias empirically at eight scales from $x \approx 10^2$ to $x \approx 10^9$, sampling up to $50{,}000$ primes at each scale. A log--log regression of $\varepsilon$ against $\log x$ yields:

\begin{equation}\label{eq:bias-decay}
\varepsilon(x) \;\approx\; \frac{c}{(\log x)^{\alpha}}, \qquad \alpha \approx 0.87, \quad c \approx 0.70.
\end{equation}

The product $\varepsilon(x) \cdot \log x$ is remarkably stable across all scales (ranging from $0.91$ to $1.04$), strongly suggesting $\alpha \approx 1$ and hence

\begin{equation}\label{eq:bias-asymptotic}
\varepsilon(x) \;\sim\; \frac{c}{\log x}
\end{equation}

for a constant $c \approx 0.97$. We note that $\varepsilon(x) \to 0$ as $x \to \infty$, consistent with the Gallagher--Poisson prediction.

\section{XOR bias absorption}
\label{sec:xor}
% ====================================================================

The following theorem is the heart of our framework: it shows that XOR sums of biased coins converge exponentially to fair.

\begin{theorem}[XOR bias absorption]\label{thm:xor-absorption}
Let $X_1, X_2, \ldots, X_m$ be independent Bernoulli random variables with $\Pr(X_i = 1) = p_i$. Let $Y = X_1 \oplus X_2 \oplus \cdots \oplus X_m$. Then
\begin{equation}\label{eq:xor-formula}
\Pr(Y = 1) = \frac{1}{2}\left(1 - \prod_{i=1}^{m}(1 - 2p_i)\right).
\end{equation}
In particular, if $p_i = \tfrac{1}{2} + \varepsilon$ for all $i$, then since $1 - 2p_i = -2\varepsilon$,
\begin{equation}\label{eq:xor-decay}
\left|\Pr(Y = 1) - \frac{1}{2}\right| = \frac{1}{2}(2|\varepsilon|)^m \;\longrightarrow\; 0
\end{equation}
exponentially fast as $m \to \infty$, for any fixed $\varepsilon \ne 0$.
\end{theorem}

\begin{proof}
By induction on $m$. For $m = 1$: $\Pr(X_1 = 1) = p_1 = \frac{1}{2}(1 - (1-2p_1))$. \checkmark

Assume the result for $m - 1$. Let $Y' = X_1 \oplus \cdots \oplus X_{m-1}$ and $q = \Pr(Y' = 1)$. By the inductive hypothesis, $1 - 2q = \prod_{i=1}^{m-1}(1 - 2p_i)$.

Since $Y = Y' \oplus X_m$ and $Y'$, $X_m$ are independent:
\begin{align*}
\Pr(Y = 1) &= \Pr(Y' = 1)\Pr(X_m = 0) + \Pr(Y' = 0)\Pr(X_m = 1) \\
&= q(1 - p_m) + (1 - q)p_m \\
&= q + p_m(1 - 2q).
\end{align*}
Therefore:
\begin{align*}
1 - 2\Pr(Y = 1) &= 1 - 2q - 2p_m(1 - 2q) = (1-2q)(1-2p_m) \\
&= \prod_{i=1}^{m}(1 - 2p_i),
\end{align*}
completing the induction.
\end{proof}

\begin{remark}
For the measured bias $\varepsilon \approx 0.07$ (primes near $10^6$), we have $2|\varepsilon| = 0.14$. After $m = 30$ XOR operations, $|{\Pr(Y=1) - 1/2}| = \frac{1}{2}(0.14)^{30} \approx 3.3 \times 10^{-26}$. Even with just $m = 5$ operations: $\frac{1}{2}(0.14)^5 \approx 2.7 \times 10^{-5}$. The convergence to fairness is extremely rapid. The Gilbreath triangle, which XORs $O(k)$ values at row~$k$, thus produces an overwhelmingly balanced first column.
\end{remark}

\section{Weak mixing of prime gap parities}
\label{sec:mixing}
% ====================================================================

Theorem~\ref{thm:xor-absorption} requires independence. The $b(n)$ values are not independent, but we show they satisfy a weaker condition.

\begin{definition}[Weak mixing]\label{def:mixing}
A binary sequence $(X_n)_{n \ge 1}$ is \emph{$(\varepsilon, \rho)$-weakly mixing} if:
\begin{enumerate}[(i)]
\item $|\Pr(X_n = 1) - 1/2| \le \varepsilon$ for all $n$, and
\item $|\Corr(X_n, X_{n+\ell})| \le A \rho^{\ell}$ for all $n$ and $\ell \ge 1$, for some constant $A > 0$ and $\rho \in (0, 1)$.
\end{enumerate}
\end{definition}

\subsection{Empirical verification for $b(n)$}

\textbf{Lag-1 correlation.} For primes up to $10^8$, we measure $\Corr(b(n), b(n+1)) \approx -0.04$ to $-0.05$. This is small but statistically significant ($p < 0.001$). The negative sign means consecutive $b$-values tend to alternate slightly.

\textbf{Lag $\ge 2$ correlations.} For all lags $\ell \ge 2$, the autocorrelation is statistically indistinguishable from zero ($p > 0.05$) at every scale tested. The absolute correlation at lag~$2$ averages $0.007$, at lag~$5$: $0.003$, at lag~$10$: $0.004$.

\textbf{$1$-dependence model.} The data strongly suggest that $b(n)$ is asymptotically $1$-dependent: the lag-$1$ correlation is small but nonzero (arising from the shared middle prime $p_{n+1}$), while all higher-lag correlations vanish. This is a considerably stronger and more precise condition than general $(\varepsilon, \rho)$-weak mixing.

\begin{figure}[ht]
\centering
\includegraphics[width=0.75\textwidth]{fig_autocorrelation.pdf}
\caption{Autocorrelation $\Corr(b(n), b(n+\ell))$ for lags $\ell = 1, \ldots, 20$ at four prime scales. The lag-$1$ correlation ($\approx -0.04$) is the only significant value; all lags $\ell \ge 2$ are statistically indistinguishable from zero, supporting the $1$-dependence model.}\label{fig:autocorrelation}
\end{figure}

\textbf{Conditional independence.} The lag-1 correlation arises from the shared middle prime $p_{n+1}$. Conditioning on $p_{n+1} \bmod 30$ reduces $|\Corr|$ from $0.040$ to $0.015$ (a $63\%$ reduction), and we expect conditioning on finer residue classes would reduce it further. This supports the view that the dependence is entirely ``local'' and disappears asymptotically.

\textbf{Connection to Gallagher's theorem.} Gallagher~\cite{Gallagher1976} proved, conditionally on Hardy--Littlewood, that prime gaps follow a Poisson process asymptotically. Poisson processes have independent increments, implying $\Corr(b(n), b(n+\ell)) \to 0$ for all $\ell \ge 1$. Our measurements are consistent with this prediction, with the residual lag-1 correlation being a finite-size effect.

\section{Main conditional theorem}
\label{sec:main}
% ====================================================================

\begin{theorem}[Conditional Gilbreath]\label{thm:main}
Assume:
\begin{enumerate}
\item[\textup{(H1)}] \textbf{Hardy--Littlewood $k$-tuple conjecture:} the number of primes $p \le x$ with $p + g$ also prime is asymptotic to $2C_2 \mathfrak{S}(g) \cdot x/(\log x)^2$.
\item[\textup{(H2)}] \textbf{Asymptotic $1$-dependence:} the sequence $b(n)$ satisfies $|\Pr(b(n)=1) - 1/2| \le \varepsilon_N$ with $\varepsilon_N \to 0$, and is asymptotically $1$-dependent: for all $\ell \ge 2$ and all $n$ sufficiently large, $b(n)$ and $b(n+\ell)$ are independent.
\item[\textup{(H3)}] \textbf{Binary absorption:} there exists $K(N)$ such that $d_k(n) \in \{0, 2\}$ for all $k \ge K(N)$ and $2 \le n \le N - k$.
\end{enumerate}
Then $d_k(1) = 1$ for all sufficiently large $k$.
\end{theorem}

\begin{proof}
The proof proceeds in seven steps.

\textit{Step 1} (Binary absorption). By~(H3), for $k \ge K_0 := K(N)$, we have $d_k(n) \in \{0, 2\}$ for all $n \ge 2$. In particular, $d_k(2) \in \{0, 2\}$.

\textit{Step 2} (Reduction to $d_k(2)$). By Proposition~\ref{prop:reduction}(a), $d_k(1)$ is non-increasing for $k \ge K_0$, and after at most finitely many additional rows, $d_k(1) \in \{1, 3\}$. Once $d_k(1) \in \{1, 3\}$, any occurrence of $d_k(2) = 2$ resets $d_{k+1}(1) = 1$ (Proposition~\ref{prop:reduction}(b)).

\textit{Step 3} (XOR dynamics). By Corollary~\ref{cor:xor}, $e_k(2) = (d_k(2)/2) \bmod 2$ follows XOR dynamics: $e_k(2) = e_{k-1}(2) \oplus e_{k-1}(3)$. Since $d_k(2) \in \{0, 2\}$ for $k \ge K_0$, we have $d_k(2) = 2 e_k(2)$. Thus $d_k(2) = 2$ if and only if $e_k(2) = 1$.

\textit{Step 4} (XOR expansion). Unrolling the XOR recurrence from Corollary~\ref{cor:xor}, we obtain
\[
e_k(2) \equiv \sum_{j=0}^{k-1} \binom{k-1}{j} b(2+j) \pmod{2}.
\]
Let $J_k = \{2+j : \binom{k-1}{j} \equiv 1 \pmod{2},\; 0 \le j \le k-1\}$. Then $e_k(2) = \bigoplus_{j \in J_k} b(j)$, i.e., $e_k(2)$ is the XOR of $|J_k|$ values from the $b$-sequence. The set $J_k \subseteq \{2, 3, \ldots, k+1\}$ always contains $2$ and $k+1$ (corresponding to $j=0$ and $j=k-1$), so $|J_k| \ge 2$. By Lucas' theorem, $|J_k| = 2^{s(k-1)}$ where $s(n)$ is the number of $1$-bits in the binary representation of~$n$. In particular, $|J_k|$ fluctuates: it equals $2$ when $k-1$ is a power of~$2$, and can be as large as $k$ when $k-1 = 2^r - 1$.

\textit{Step 5} (XOR bias absorption via $1$-dependence). We treat two cases based on $|J_k|$.

\textit{Case~1:} $|J_k| \ge 3$. Under hypothesis~(H2), $b(n)$ is asymptotically $1$-dependent, meaning variables at distance $\ge 2$ are independent. From $J_k$ we extract a non-adjacent subset $M_k \subseteq J_k$ (no two elements consecutive) by greedy selection, giving $|M_k| \ge \lceil |J_k|/2 \rceil \ge 2$. The variables $\{b(j)\}_{j \in M_k}$ are mutually independent under $1$-dependence. By conditioning on any fixed realization of $C_k = J_k \setminus M_k$, the parity of $e_k(2)$ depends on the XOR of the independent variables in $M_k$. Applying Theorem~\ref{thm:xor-absorption} directly (with no remainder term):
\[
\left|\Pr(e_k(2) = 1) - \frac{1}{2}\right| \le \frac{1}{2}\prod_{j \in M_k} 2|\varepsilon_j|.
\]
By~(H2), $|\varepsilon_j| \le \varepsilon_N \to 0$, so each factor $2|\varepsilon_j| < 1$. When $|M_k| \ge 2$ and $\varepsilon_N$ is small, this product is at most $(2\varepsilon_N)^2$, which is small.

\textit{Case~2:} $|J_k| = 2$, i.e., $J_k = \{2, k+1\}$. Here $e_k(2) = b(2) \oplus b(k+1)$. Since $k+1 \ge 4$, the distance between elements is $k-1 \ge 2$, so by~(H2) the two variables are independent. By Theorem~\ref{thm:xor-absorption}:
\[
\left|\Pr(e_k(2) = 1) - \frac{1}{2}\right| = \frac{1}{2} \cdot 2|\varepsilon_2| \cdot 2|\varepsilon_{k+1}| = 2|\varepsilon_2 \varepsilon_{k+1}|.
\]
For $\varepsilon_j \approx 0.07$, this gives a bias of approximately $0.01$, so $\Pr(e_k(2) = 1) \approx 0.49$---still bounded away from zero.

In both cases, $\Pr(e_k(2) = 1) \ge 1/2 - (2\varepsilon_N)^2$ for all sufficiently large $k$. As $\varepsilon_N \to 0$, this approaches~$1/2$.

\textit{Step 6} (Frequency of reset pulses). By Step~5, $\Pr(d_k(2) = 2) = \Pr(e_k(2) = 1) \ge 1/2 - (2\varepsilon_N)^2$ for all $k \ge K_0$. Since $\varepsilon_N \to 0$, this probability approaches~$1/2$. In particular, $\Pr(d_k(2) = 2)$ is bounded away from zero uniformly in~$k$, so the expected waiting time between consecutive resets is bounded.

\textit{Step 7} (Conclusion). By Step~2, $d_k(1)$ is non-increasing once binary absorption holds, and after finitely many rows $d_k(1) \in \{1, 3\}$. Once $d_k(1) \in \{1, 3\}$, every occurrence of $d_k(2) = 2$ resets $d_{k+1}(1) = 1$ (Proposition~\ref{prop:reduction}(b)). By Step~6, these resets occur with positive density bounded away from zero. Between resets, $d_k(2) = 0$ preserves the value (Proposition~\ref{prop:reduction}(c)). Therefore, once $d_k(1)$ reaches~$1$ via the descent mechanism, it remains at~$1$ permanently. Thus $d_k(1) = 1$ for all sufficiently large~$k$.
\end{proof}

\begin{remark}[On the gap to an unconditional proof]
The three hypotheses address different aspects:
\begin{itemize}
\item (H1) is the Hardy--Littlewood conjecture, one of the central open problems in number theory. It implies the bias formula in Section~\ref{sec:bias}.
\item (H2) is a quantitative form of Gallagher's result~\cite{Gallagher1976}, specialized to $1$-dependence. Our computational evidence (Section~\ref{sec:mixing}) strongly supports it: lag-$1$ correlation is small but nonzero (due to the shared middle prime), while correlations for all lags $\ell \ge 2$ are statistically indistinguishable from zero at every scale tested up to $10^8$.
\item (H3) extends Chase's Theorem~2~\cite{Chase2020} from random sequences (with gaps $\le O(\log\log M)$) to primes (with gaps $\le O(\log p)$). This is the most challenging gap to close.
\end{itemize}
\end{remark}
